00265   if( ( m_scene->antCount() + ui->deadLineEdit->text().toInt() ) < ui->nrAntsSpinBox->value() )
00266   {
00267     if( m_scene->spawnPoints().size() > 0 )
00268     {
00269       /* Choose a random spawn point if there are more than one. */
00270       int point = qrand() % m_scene->spawnPoints().size();
00271       m_scene->registerAnt( m_scene->spawnPoints().at( point )->centre() );
00272     }
00273   }
00274   else
00275   {
00276     /* Once the desired nr of ants have been created, disconnect the timer from this slot. */
00277     disconnect( m_synchTimer, SIGNAL( timeout() ), this, SLOT( spawn() ) );
