00104 void AntWorld::registerPheromone( const AntPosition& position, AntPheromone::PheromoneType type, SharedAntPtr& ant )
00105 {

00113       /* Don't change the type if the pheromone is a hazard pheromone. */

00116         pheromone->setPheromoneType( type );

00118     }
00119     else
00120     {
00121       /* For now this is sufficient, if other types of smart pheromone is going to be

00123       if ( ( type == AntPheromone::Hazard && m_smartPheromoneEnabled ) ||
00124            ( type != AntPheromone::Hazard && m_pheromoneEnabled ) )
00125       {
00126         pheromone = createPheromone( position, type );
00127         pheromone->setEvaporationRate( m_evaporationRate );

00129 
00130         /* Register the pheromone with the tile it was dropped on. */
00131         AntWorldTile* tile = findTile( position );

00133       }
00134     }

00138       pheromone->registerAnt( ant );

00199 void AntWorld::updateAnts()
00200 {
00201   m_foragingAnts = 0;
00202   m_gatheringAnts = 0;
00203 
00204   for( auto& ant : m_ants )
00205   {
00206     /* Take the next step first (we know where we are, but don't know where we are going). */
00207     ant->advance();
00208 
00209     if( ant->droppedPheromone() ) registerPheromone( ant->position(), ant->droppedPheromoneType(), ant );
00210 
00211     doGatheringAntLogic( ant );
00212     doAntPheromoneDeregistration( ant );
00213   }
00214 
00215   doDeadAntLogic();
00216   doForagingAntLogic();
00217 }
