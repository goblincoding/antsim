00053 void GraphicsAntWorldScene::constructWorldGrid()
00054 {
00055   reset();
00056 
00057   qreal totalWidth = sceneRect().width() * ( 1.0 - marginPerc );
00058   qreal totalHeight = sceneRect().height() * ( 1.0 - marginPerc );
00059 
00060   qreal x( ( sceneRect().width() - totalWidth ) / 2 );
00061 
00062   while( x < totalWidth )
00063   {
00064     qreal y( ( sceneRect().height() - totalHeight ) / 2 );
00065 
00066     while( y < totalHeight )
00067     {
00068       m_currentTileRect = QRectF( x, y, AntConfig::TileSize, AntConfig::TileSize );
00069       registerWorldTile( calculateCentreCoordinate( m_currentTileRect ), AntWorldTile::Wall );
00070       y += AntConfig::TileSize;
00071     }
00072 
00073     x += AntConfig::TileSize;
00074   }
00075 }

00105 std::vector< const AntWorldTile* > GraphicsAntWorldScene::neighbours( const AntPosition& position ) const
00106 {
00107   QPointF pointF( qreal( position.x() ), qreal( position.y() ) );
00108 
00109   /* Since we're working in a 2D grid, we can get all 4 neighbour tiles by offsetting
00110    * the relevant coordinate with the amount equal to the square edge of a tile (which
00111    * should place us roughly near the centre of whichever neighbour we're searching for. */
00112   QPointF northNeighbour( pointF + QPointF( 0.0, AntConfig::TileSize ) );
00113   QPointF southNeighbour( pointF - QPointF( 0.0, AntConfig::TileSize ) );
00114   QPointF westNeighbour( pointF + QPointF( AntConfig::TileSize, 0.0 ) );
00115   QPointF eastNeighbour( pointF - QPointF( AntConfig::TileSize, 0.0 ) );
00116 
00117   QList< QPointF > neighboursPoints = QList< QPointF >() << northNeighbour << southNeighbour << westNeighbour << eastNeighbour;
00118   std::vector< const AntWorldTile* > neighbours;
00119 
00120   for( int i = 0; i < neighboursPoints.size(); ++i )
00121   {
00122     QList< QGraphicsItem* > itemList = items( neighboursPoints.at( i ) );
00123 
00124     for( int j = 0; j < itemList.size(); ++j )
00125     {
00126       GraphicsWorldTile* tile = dynamic_cast< GraphicsWorldTile* >( itemList.at( j ) );
00127       if( tile ) neighbours.push_back( tile );
00128     }
00129   }
00130 
00131   return neighbours;
00132 }

00211 void GraphicsAntWorldScene::changeItemType( const QPointF& point )
00212 {
00213   if( m_type == GraphicsWorldTile::None )
00214   {
00215     return;
00216   }
00217 
00218   QList< QGraphicsItem* > itemList = items( point );
00219 
00220   for( int i = 0; i < itemList.size(); ++i )
00221   {
00222     GraphicsWorldTile* tile = dynamic_cast< GraphicsWorldTile* >( itemList.at( i ) );
00223 
00224     if( tile )
00225     {
00226       if( tile->tileType() == GraphicsWorldTile::Spawn )
00227       {
00228         m_spawnPoints.removeAll( tile );

00230 
00231       if( m_type == GraphicsWorldTile::Spawn )
00232       {
00233         m_spawnPoints.append( tile );
00234       }
