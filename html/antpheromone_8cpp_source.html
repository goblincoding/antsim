00049 void AntPheromone::update()
00050 {
00051   m_ants.erase( std::remove_if( std::begin( m_ants ), std::end( m_ants ),
00052                                 []( WeakAntPtr& ant ){ return ant.expired(); } ),
00053                 std::end( m_ants ) );
00054 
00055   updateGraphics();
00056 
00057   /* Hazards will remain active for the duration of the run and will not evaporate. */
00058   if( m_type == Hazard )
00059   {
00060     m_strength = 1.0;
00061   }
00062   else
00063   {
00064     auto antPheromoneSum( 0.0 );
00065 
00066     for( auto& ant : m_ants )
00067     {
00068       SharedAntPtr bot = ant.lock();
00069       if( bot ) antPheromoneSum += bot->droppedPheromoneStrength();
00070     }
00071 
00072     m_strength = ( 1.0 - m_evaporationRate ) * m_strength + antPheromoneSum;
00073   }
00074 }
