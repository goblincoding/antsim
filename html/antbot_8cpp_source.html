00067 void AntBot::advance()
00068 {
00069   m_doPheromoneDeregistration = false;
00070 
00071   /* Ant states turn to "DroppingPheromone" as soon as they have found the food source for
00072    * the first time.  This enables them to continue dropping pheromone on the tiles leading
00073    * to the food source, but also to "restart" the foraging experience (so that they do not
00074    * end up continuously running between nest and food). */
00075   switch( m_state )
00076   {
00077     case Gathering:
00078       gather();
00079       break;
00080     case Foraging:            // deliberate fall-through
00081     case DroppingPheromone:
00082       forage();
00083       break;
00084     case Dead:
00085       break;
00086   }
00087 }

00218 const AntPosition& AntBot::determineNextPosition()
00219 {
00220   /* If, for whatever reason, a tile has no neighbours (you never know how the world
00221    * might be constructed), return the current position. */
00222   if( m_neighbours.size() == 0 ) return m_position;
00223 
00224   std::vector< int > probabilities = calculateNeighbourProbabilities();
00225 
00226   /* Calculate the normalisation factor (sum of probabilities must ultimately add to 1.0). */
00227   double normalisationFactor = static_cast< double >( AntConfig::ProbabilityRange ) /
00228                                std::accumulate( probabilities.begin(), probabilities.end(), 0 );
00229 
00230   /* Now we calculate the normalised probabilities for each tile and stick these normalised values
00231    * into a map against their original, pre-sorted indices (otherwise we will not know which index
00232    * went with which tile initially). */
00233   std::multimap< int /*value*/, int /*index*/ > indices;
00234 
00235   for( std::vector< int >::size_type i = 0; i < probabilities.size(); ++i )
00236   {
00237     probabilities[ i ] = static_cast< int >( probabilities[ i ] * normalisationFactor );
00238     indices.insert( std::make_pair( probabilities[ i ], i ) );
00239   }
00240 
00241   /* Sort the values from lowest to highest, randomly select a value in the
00242    * probability range and see in which interval the selected probability falls. */
00243   std::sort( probabilities.begin(), probabilities.end() );
00244   int selectedProbability = rand() % AntConfig::ProbabilityRange + 1;   // don't want to select 0 prob (disallowed) tiles
00245 
00246   std::vector< int >::size_type index = 0;
00247 
00248   while( index < probabilities.size() - 1 )
00249   {
00250     if( selectedProbability > probabilities[ index ] )
00251     {
00252       ++index;
00253     }
00254     else
00255     {
00256       break;
00257     }
00258   }
00259 
00260   /* Use the convenience mapping to obtain the index of the tile associated with the
00261    * selected probability and use that index (remember the indices of "probabilities"
00262    * correspond to those of m_neighbours) to find the selected tile's centre position. */
00263   return m_neighbours[ ( indices.find( probabilities[ index ] ) )->second ]->centre();
00264 }

00268 int AntBot::calculateTileProbabilityMax( const AntWorldTile* tile ) const
00269 {
00270   /* Skip all tiles where ants have gone...and died...shame...*/
00271   if( tile->tilePheromoneType() == AntPheromone::Hazard ) return 0;
00272 
00273   switch( tile->tileType() )
00274   {
00275     case AntWorldTile::Path:    // deliberate fall-through
00276     case AntWorldTile::Spawn:   // deliberate fall-through
00277     case AntWorldTile::Food:
00278       {
00279         /* Prefer nodes moving away from the last visited node by penalising recently visited nodes. */
00280         double finalRange = AntConfig::ProbabilityRange;
00281         if( m_graph->recentlyVisited( tile->centre() ) ) finalRange /= AntConfig::RecentlyVisitedPenalty;
00282 
00283         double probabilityMax = ( tile->tilePheromoneStrength() / m_neighbourPheromoneSum ) * finalRange;
00284         if( probabilityMax < 1.0 ) probabilityMax = rand() % static_cast< int >( finalRange ) + 1;  // +1 since we can't return 0 values for non-wall/non-hazard tiles
00285 
00286         return static_cast< int >( probabilityMax );
00287       }
00288     case AntWorldTile::Wall:    // deliberate fall-through
00289     case AntWorldTile::Hazard:  // deliberate fall-through
00290     case AntWorldTile::None:
00291       return 0;
00292   }
00293 
00294   return 0;
00295 }

00299 std::vector< int > AntBot::calculateNeighbourProbabilities()
00300 {
00301   m_neighbourPheromoneSum = std::accumulate( std::begin( m_neighbours ), std::end( m_neighbours ), 0.0,
00302                                              []( double sum, const AntWorldTile*& tile ) { return sum + tile->tilePheromoneStrength(); } );
00303 
00304   /* First we determine the selection probability maximum for each tile. The indices
00305    * of the probability maximums will coincide with the indices of their respective
00306    * tiles in m_neighbours. */
00307   std::vector< int > probabilities;
00308   for( auto& neighbour : m_neighbours ) probabilities.push_back( calculateTileProbabilityMax( neighbour ) );
00309 
00310   /* Introduce a "shake" factor to distinguish duplicate values from each other. The range is
00311    * a bit of an arbitrary calculation since the value must be relatively small compared to
00312    * the entire probability range, yet cannot be fixed in case the parameters change. */
00313   int shakeRange = AntConfig::ProbabilityRange / AntConfig::RecentlyVisitedPenalty;
00314 
00315   for( auto probability : probabilities )
00316   {
00317     /* We don't want to shake 0 values since these must remain 0 (Walls and Hazards should
00318      * definitely not be traversable). */
00319     if( probability > 0 )
00320     {
00321       /* Check if we have duplicates. */
00322       if( std::count( std::begin( probabilities ), std::end( probabilities ), probability ) > 1 )
00323       {
00324         /* Find the duplicates of this specific probability and apply the shake
00325           factor to all of them. */
00326         auto it = std::find( std::begin( probabilities ), std::end( probabilities ), probability );
00327 
00328         while( it != std::end( probabilities ) )
00329         {
00330           *it += rand() % shakeRange;
00331           it = std::find( it + 1, std::end( probabilities ), probability );
00332         }
00333       }
00334     }
00335   }
00336 
00337   return probabilities;
00338 }

00354 void AntBot::gather()
00355 {
00356   if( m_returningToSource )
00357   {
00358     if( m_stepsFromTarget < m_shortestPathLength )
00359     {
00360       auto position = m_graph->shortestPath()[ m_stepsFromTarget ];
00361       updatePosition( position );
00362       ++m_stepsFromTarget;
00363     }
00364     else
00365     {
00366       m_stepsFromTarget = 0;
00367       m_returningToSource = false;
00368     }
00369 
00370     m_droppedPheromone = true;
00371   }
00372   else
00373   {
00374     setAntState( DroppingPheromone );
00375   }

00380 bool AntBot::foundFood( const AntWorldTile* tile )
00381 {
00382   if( tile->tileType() == AntWorldTile::Food )
00383   {
00384     m_graph->setFoundTarget( tile->centre() );
00385 
00386     /* See if the new source/target path is shorter than the previous before we
00387      * change the state, etc to avoid switching to a worse solution. */
00388     int newShortestPathLength = static_cast< int >( m_graph->shortestPath().size() );
00389 
00390     if( m_shortestPathLength > newShortestPathLength )
00391     {
00392       setAntState( Gathering );
00393       m_shortestPathLength = newShortestPathLength;
00394       m_pheromoneStrength = ( 1.0 / static_cast< double >( m_shortestPathLength ) );
00395       m_pheromoneType = AntPheromone::Found;
00396       m_returningToSource = true;
00397       m_deRegisteredPheromones.clear();
00398       m_deRegisteredPheromones = m_pheromones;
00399       m_pheromones.clear();
00400       m_doPheromoneDeregistration = true;
00401       return true;
00402     }
00403   }
00404 
00405   return false;
