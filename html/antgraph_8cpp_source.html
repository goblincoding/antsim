00044 void AntGraph::addNode( const AntPosition& position )
00045 {
00046   /* Keep track of two separate lists of nodes, the first is a list of ALL the nodes
00047    * ever visited and the second that of the nodes on the current shortest path.  The
00048    * reason for keeping the first list is that, even though ants may double back on
00049    * themselves, we'd still like to know if the nodes they travel along on the next
00050    * "iteration" of their travels has been previously visited in order to favour nodes
00051    * that have never been visited at all (e.g. go down a new fork, rather than simply
00052    * traversing the previous loop that ended in failure). */
00053   if( std::none_of( std::begin( m_nodes ), std::end( m_nodes ),
00054                     [ &position ]( AntPosition pos ) { return pos == position; } ) )
00055   {
00056     m_nodes.push_back( position );
00057   }
00058   else
00059   {
00060     /* If a node is already in the list, it means the ant has looped back on itself,
00061      * in other words, all the nodes encountered since we last visited this particular
00062      * node has been for naught and can be deleted.  This way, we ensure that the route
00063      * the ant travels between the starting point and the final destination remains the
00064      * shortest route for its particular graph. */
00065     auto it = std::find( std::begin( m_shortestPath ), std::end( m_shortestPath ), position );
00066 
00067     /* Remove the pointless nodes (the loop) from the vector. */
00068     m_shortestPath.resize( it - std::begin( m_shortestPath ) );
00069   }
00070 
00071   /* Regardless of what happens, add the node to the end of the shortest path (if the
00072    * node had been previously encountered, it would have been removed in the "else" part
00073    * above and will need to be re-added). */
00074   m_shortestPath.push_back( position );
00075 
00076   /* We don't want the same position to occur multiple times in the recently
00077    * visited list since this will skew the results when "recentlyVisited" is called. */
00078   if( !recentlyVisited( position ) )
00079   {
00080     if( m_recentlyVisited.size() < m_maxNodesRemembered )
00081     {
00082       m_recentlyVisited.push_back( position );
00083     }
00084     else
00085     {
00086       m_recentlyVisited.pop_front();
00087       m_recentlyVisited.push_back( position );
00088     }
00089   }
00090 }
